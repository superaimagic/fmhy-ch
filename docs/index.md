<script setup>
import { onMounted } from 'vue'

onMounted(() => {
  // 首选的可爱模式偏好
  var preferredKawaii
  try {
    preferredKawaii = localStorage.getItem('uwu')
  } catch (err) {}
  // 从 URL 参数获取可爱模式设置
  const urlParams = new URLSearchParams(window.location.search)
  const kawaii = urlParams.get('uwu')
  // 启用可爱模式
  const setKawaii = () => {
    const images = document.querySelectorAll('.VPImage.image-src')
    images.forEach((img) => {
      img.src = '/logo-uwu.svg'
    })
  }
  // 重置为默认图标
  const resetKawaii = () => {
    const images = document.querySelectorAll('.VPImage.image-src')
    images.forEach((img) => {
      img.src = '/test.png'
    })
  }
  if (kawaii === 'true') {
    try {
      localStorage.setItem('uwu', true)
    } catch (err) {}
    console.log('可爱模式已启用。使用 "?uwu=false" 禁用。');
    setKawaii()
  } else if (kawaii === 'false') {
    try {
      localStorage.removeItem('uwu', false)
    } catch (err) {}
    resetKawaii()
  } else if (preferredKawaii) {
    setKawaii()
  }

  // 点击英雄图片 5 次切换可爱模式
  let clickCount = 0;
  const heroImage = document.querySelector('.VPImage.image-src');
  
  const handleClick = () => {
    clickCount += 1;
    if (clickCount === 5) {
      const isKawaii = localStorage.getItem('uwu') === 'true';
      if (isKawaii) {
        localStorage.removeItem('uwu');
        resetKawaii();
        console.log('可爱模式已禁用。');
      } else {
        localStorage.setItem('uwu', true);
        setKawaii();
        console.log('点击 5 次后启用可爱模式。');
      }
      clickCount = 0;
    }
  };

  if (heroImage) {
    heroImage.addEventListener('click', handleClick);
  }
})
</script>
